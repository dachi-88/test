HTML / JavaScript

<script>
  // ფუნქცია onCallBack გამოიძახება ავტორიზაციის დასრულების შემდეგ
  // data პარამეტრში ფუნქციას გადმოეცემა ავტორიზაციის შედეგი
  function onCallBack(data) {
    if (data.Authenticated) {
    alert(‘თქვენ წარმატებით გაიარეთ ავტორიზაცია RS-ზე ’ + ‘ავტორიზაციის კოდია: ’ + data.AuthKey);

    // აქ ჩაწერეთ კოდი საიდანაც იძახებთ თქვენს სერვერს და გადასცემთ data.AuthKey-ს
    // data.AuthKey - ში მოდის ავტორიზაციის კოდი, რომელიც
    // SecretKey-სთან ერთად პარამეტრად უნდა გადასცეთ ჩვენს სერვისს. 
    // სერვისი აუცილებლად უნდა გამოიძახოთ სერვერის მხრიდან!
    }
    else {
      alert(‘თქვენ ვერ გაიარეთ ავტორიზაცია RS-ზე ’ + ‘მიზეზი: ’ + data.Response);
    }
  }
</script>

// rs-login არის RS პორტალზე შესვლის ღილაკი, მასზე კლიკით გამოდის RS ავტორიზაციის პოპაპი;
// ავტორიზაციის დასრულების (წარმატებით ან წარუმატებლად) შემდეგ ავტომატურად იძახება

სერვისის მხარე

ღია ავტორიზაცია სერვისის მხარე

  protected void callRsService(string AuthKey)
    {
      //Secret Key for your app
      string SecretKey = "{Your-Secret-Key}";

      //Call Service with Secret Key and Auth Key (Returned in client)
      string RequestString = string.Format("https://eservices.rs.ge/WebServices/oAuth.ashx" +
      "?SecretKey={0}&AuthKey={1}", SecretKey, AuthKey);
      WebRequest request = WebRequest.Create(RequestString);
      request.Credentials = CredentialCache.DefaultCredentials;
      WebResponse response = request.GetResponse();
      Console.WriteLine(((HttpWebResponse)response).StatusDescription);
      Stream dataStream = response.GetResponseStream();
      StreamReader reader = new StreamReader(dataStream);
      string rsResponse = reader.ReadToEnd();
      
      // rsResponse contains XML about authorized user
      Console.WriteLine(rsResponse);
      reader.Close();
      response.Close();
    }
